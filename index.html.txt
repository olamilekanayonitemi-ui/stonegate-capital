                    <h3 className="text-xl font-bold text-gray-800 mb-4">Employment & Income</h3>

                      <label className="block text-sm font-medium text-gray-700 mb-2">Employer Name *</label>

                        <label className="block text-sm font-medium text-gray-700 mb-2">Monthly Income *</label>

                          type="number" 
                          name="monthlyIncome"
                          value={formData.monthlyIncome}

                          className="w-full border-2 border-gray-300 rounded-lg px-4 py-3 focus:border-blue-600 focus:outline-none transition"

                          type="number" 
                          name="additionalIncome"
                          value={formData.additionalIncome}
                          onChange={handleInputChange}
                          placeholder="0"
                          className="w-full border-2 border-gray-300 rounded-lg px-4 py-3 focus:border-blue-600 focus:outline-none transition"
                        />
                      </div>
                    </div>
                  </div>
                )}

                {/* Step 4: Documents & Consent */}

                            </div>
                            <button onClick={() => removeFile('idDocument')} className="text-red-600 hover:text-red-700">

                            </button>
                          </div>

                      </div>

                      <div className="border-2 border-dashed border-gray-300 rounded-lg p-6 hover:border-blue-600 transition">
                        <label className="block text-sm font-semibold text-gray-700 mb-3">
                          Proof of Income * (Pay stub, Bank statement, or Tax return)
                        </label>
                        {!uploadedFiles.proofOfIncome ? (
                          <label className="flex flex-col items-center cursor-pointer">
                            <Upload className="w-12 h-12 text-gray-400 mb-2" />
                            <span className="text-sm text-gray-600">Click to upload or drag and drop</span>
                            <span className="text-xs text-gray-500 mt-1">PNG, JPG, PDF up to 10MB</span>
                            <input 
                              type="file" 
                              accept=".pdf,.jpg,.jpeg,.png"
                              onChange={(e) => handleFileUpload(e, 'proofOfIncome')}

                            />
                          </label>
                        ) : (
                          <div className="flex items-center justify-between bg-green-50 p-4 rounded-lg">
                            <div className="flex items-center">
                              <CheckCircle className="w-5 h-5 text-green-600 mr-2" />
                              <span className="text-sm font-medium">{uploadedFiles.proofOfIncome.name}</span>
                            </div>
                            <button onClick={() => removeFile('proofOfIncome')} className="text-red-600 hover:text-red-700">
                              <X className="w-5 h-5" />
                            </button>
                          </div>
                        )}
                      </div>

                      <div className="border-2 border-dashed border-gray-300 rounded-lg p-6 hover:border-blue-600 transition">
                        <label className="block text-sm font-semibold text-gray-700 mb-3">
                          Proof of Address * (Utility bill or Bank statement)
                        </label>
                        {!uploadedFiles.proofOfAddress ? (
                          <label className="flex flex-col items-center cursor-pointer">
                            <Upload className="w-12 h-12 text-gray-400 mb-2" />
                            <span className="text-sm text-gray-600">Click to upload or drag and drop</span>
                            <span className="text-xs text-gray-500 mt-1">PNG, JPG, PDF up to 10MB</span>
                            <input 
                              type="file" 
                              accept=".pdf,.jpg,.jpeg,.png"
                              onChange={(e) => handleFileUpload(e, 'proofOfAddress')}
                              className="hidden"
                            />
                          </label>
                        ) : (
                          <div className="flex items-center justify-between bg-green-50 p-4 rounded-lg">
                            <div className="flex items-center">
                              <CheckCircle className="w-5 h-5 text-green-600 mr-2" />
                              <span className="text-sm font-medium">{uploadedFiles.proofOfAddress.name}</span>
                            </div>
                            <button onClick={() => removeFile('proofOfAddress')} className="text-red-600 hover:text-red-700">
                              <X className="w-5 h-5" />
                            </button>
                          </div>
                        )}
                      </div>
                    </div>

                    <div className="bg-yellow-50 border-2 border-yellow-200 rounded-lg p-6 mt-6">
                      <label className="flex items-start cursor-pointer">
                        <input 
                          type="checkbox"
                          checked={agreedToTerms}
                          onChange={(e) => setAgreedToTerms(e.target.checked)}
                          className="mt-1 mr-3 w-5 h-5 accent-blue-600"
                        />
                        <span className="text-sm text-gray-700">
                          <strong>I agree and understand that this is a loan that must be repaid.</strong> I have read and agree to the Terms of Service and Privacy Policy. I authorize Stonegate Capital to verify my information and contact me regarding my application.
                        </span>
                      </label>
                    </div>
                  </div>
                )}

                {/* Navigation Buttons */}
                <div className="flex justify-between mt-8 pt-6 border-t">
                  {currentStep > 1 && (
                    <button 
                      onClick={prevStep}
                      className="flex items-center px-6 py-3 border-2 border-gray-300 rounded-lg font-semibold text-gray-700 hover:bg-gray-50 transition"
                    >
                      <ArrowLeft className="w-5 h-5 mr-2" />
                      Previous
                    </button>
                  )}
                  
                  {currentStep < 4 ? (
                    <button 
                      onClick={nextStep}
                      className="ml-auto flex items-center px-8 py-3 bg-blue-600 text-white rounded-lg font-semibold hover:bg-blue-700 transition shadow-lg"
                    >
                      Continue
                      <ArrowRight className="w-5 h-5 ml-2" />
                    </button>
                  ) : (
                    <button 
                      onClick={handleSubmit}
                      className="ml-auto flex items-center px-8 py-3 bg-green-600 text-white rounded-lg font-bold hover:bg-green-700 transition shadow-lg"
                    >
                      Submit Application
                      <CheckCircle className="w-5 h-5 ml-2" />
                    </button>
                  )}
                </div>
              </>
            ) : (
              <div className="text-center py-12">
                <div className="bg-green-100 w-24 h-24 rounded-full flex items-center justify-center mx-auto mb-6">
                  <CheckCircle className="w-16 h-16 text-green-600" />
                </div>
                <h3 className="text-3xl font-bold mb-4 text-gray-800">Application Received!</h3>
                <p className="text-lg text-gray-600 mb-2">
                  Thank you for choosing Stonegate Capital.
                </p>
                <p className="text-gray-600 mb-8">
                  We've received your application and will review it within 24-48 hours.
                  You'll receive an email confirmation shortly.
                </p>
                <div className="bg-blue-50 p-6 rounded-lg mb-8 max-w-md mx-auto border border-blue-200">
                  <p className="text-sm text-gray-700 mb-2">
                    <strong className="text-blue-600">Application ID:</strong> SG-{Math.floor(Math.random() * 1000000)}
                  </p>
                  <p className="text-sm text-gray-700">
                    <strong className="text-blue-600">Submitted:</strong> {new Date().toLocaleString()}
                  </p>
                </div>
                <div className="space-y-4">
                  <h4 className="font-semibold text-gray-800">What happens next?</h4>
                  <div className="max-w-md mx-auto text-left space-y-3">
                    <div className="flex items-start">
                      <div className="bg-blue-100 rounded-full p-2 mr-3">
                        <span className="text-blue-600 font-bold text-sm">1</span>
                      </div>
                      <p className="text-sm text-gray-600 pt-1">We'll verify your information and documents</p>
                    </div>
                    <div className="flex items-start">
                      <div className="bg-blue-100 rounded-full p-2 mr-3">
                        <span className="text-blue-600 font-bold text-sm">2</span>
                      </div>
                      <p className="text-sm text-gray-600 pt-1">You'll receive a decision via email within 24-48 hours</p>
                    </div>
                    <div className="flex items-start">
                      <div className="bg-blue-100 rounded-full p-2 mr-3">
                        <span className="text-blue-600 font-bold text-sm">3</span>
                      </div>
                      <p className="text-sm text-gray-600 pt-1">If approved, funds will be disbursed within 3-5 business days</p>
                    </div>
                  </div>
                </div>
                <button 
                  onClick={() => {
                    setSubmitted(false);
                    setCurrentStep(1);
                    setAgreedToTerms(false);
                    setFormData({
                      firstName: '',
                      lastName: '',
                      email: '',
                      phone: '',
                      dateOfBirth: '',
                      ssn: '',
                      address: '',
                      city: '',
                      state: '',
                      zipCode: '',
                      loanType: 'personal',
                      loanAmount: '',
                      loanPurpose: '',
                      loanTerm: '',
                      employmentStatus: '',
                      employer: '',
                      monthlyIncome: '',
                      additionalIncome: '',
                      idDocument: null,
                      proofOfIncome: null,
                      proofOfAddress: null
                    });
                    setUploadedFiles({
                      idDocument: null,
                      proofOfIncome: null,
                      proofOfAddress: null
                    });
                    setActiveTab('home');
                  }}
                  className="mt-8 bg-blue-600 text-white px-8 py-3 rounded-lg font-semibold hover:bg-blue-700 transition shadow-lg"
                >
                  Return to Home
                </button>
              </div>
            )}
          </div>
        </div>
      )}

      {/* Footer */}
      <footer className="bg-gray-900 text-white mt-20">
        <div className="max-w-7xl mx-auto px-4 py-12 sm:px-6 lg:px-8">
          <div className="grid md:grid-cols-4 gap-8 mb-8">
            <div className="col-span-2">
              <div className="flex items-center space-x-3 mb-4">
                <div className="bg-blue-600 p-2 rounded-lg">
                  <DollarSign className="w-6 h-6 text-white" />
                </div>
                <div>
                  <span className="text-2xl font-bold">Stonegate Capital</span>
                  <p className="text-sm text-gray-400">Structured Loan Funding Community</p>
                </div>
              </div>
              <p className="text-gray-400 text-sm mb-4">
                Empowering individuals and businesses with accessible, transparent lending solutions. 
                Your trusted partner for financial growth.
              </p>
              <div className="flex space-x-4">
                <a href="https://twitter.com" target="_blank" rel="noopener noreferrer" className="bg-gray-800 p-3 rounded-lg hover:bg-gray-700 transition">
                  <Twitter className="w-5 h-5" />
                </a>
              </div>
            </div>
            
            <div>
              <h4 className="font-bold mb-4">Company</h4>
              <ul className="space-y-2 text-gray-400 text-sm">
                <li><a href="#" className="hover:text-white transition">About Us</a></li>
                <li><a href="#" className="hover:text-white transition">Contact</a></li>
                <li><a href="#" className="hover:text-white transition">Careers</a></li>
                <li><a href="#" className="hover:text-white transition">Blog</a></li>
              </ul>
            </div>
            
            <div>
              <h4 className="font-bold mb-4">Legal</h4>
              <ul className="space-y-2 text-gray-400 text-sm">
                <li><a href="#" className="hover:text-white transition">Privacy Policy</a></li>
                <li><a href="#" className="hover:text-white transition">Terms of Service</a></li>
                <li><a href="#" className="hover:text-white transition">Loan Agreement</a></li>
                <li><a href="#" className="hover:text-white transition">Compliance</a></li>
              </ul>
            </div>
          </div>
          
          <div className="border-t border-gray-800 pt-8 text-center">
            <p className="text-gray-400 text-sm">

            </p>
            <p className="text-gray-500 text-xs mt-2">

            </p>
